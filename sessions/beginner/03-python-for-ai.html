Why Python? Why Not JavaScript or Java?

Python became the language of AI because of three things: readable syntax, an unbeatable ecosystem (NumPy, Pandas, PyTorch, scikit-learn), and a massive community. When you need to use a new ML library, the docs, tutorials, and Stack Overflow answers are all in Python.

Lists and Loops

Lists are ordered, mutable collections. In AI, you'll use lists for datasets, batches of text, token sequences, and API message history.

# Create and access lists
messages = ["Hello!", "What is AI?", "Explain transformers"]
print(messages[0])          # "Hello!"    (0-indexed)
print(messages[-1])         # "Explain transformers"  (last item)
print(len(messages))        # 3

# Append, extend, remove
messages.append("Thanks!")
messages.extend(["Bye", "See you"])
messages.remove("Thanks!")

# List comprehensions — used EVERYWHERE in data processing
texts = ["  hello  ", "  world  ", "  ai  "]
cleaned = [t.strip() for t in texts]
# cleaned = ["hello", "world", "ai"]

# Filter with comprehension
long_texts = [t for t in texts if len(t.strip()) > 3]

# Looping
for i, msg in enumerate(messages):
    print(f"[{i}] {msg}")

# Slicing — critical for working with context windows
recent = messages[-3:]      # last 3 messages
first_half = messages[:len(messages)//2]

Dictionaries

Dictionaries are key-value stores. LLM API messages, model configs, and JSON data all come back as Python dicts.

# The structure of an OpenAI API message
message = {
    "role": "user",
    "content": "Explain neural networks in 2 sentences"
}

# Access values
print(message["role"])      # "user"
print(message.get("model", "gpt-4o"))  # safe access with default

# Nested dicts — common in API responses
api_response = {
    "id": "chatcmpl-abc123",
    "model": "gpt-4o",
    "usage": {
        "prompt_tokens": 15,
        "completion_tokens": 48,
        "total_tokens": 63
    },
    "choices": [
        {"message": {"role": "assistant", "content": "Neural networks are..."}}
    ]
}

# Navigate nested structure
content = api_response["choices"][0]["message"]["content"]
tokens_used = api_response["usage"]["total_tokens"]
print(f"Response: {content[:50]}...")
print(f"Tokens used: {tokens_used}")

# Dict comprehension
token_counts = {"input": 15, "output": 48, "total": 63}
doubled = {k: v * 2 for k, v in token_counts.items()}

Functions

Functions let you package reusable logic. Almost every AI utility you write will be a function.

import openai

def ask_llm(prompt: str, model: str = "gpt-4o", max_tokens: int = 500) -> str:
    """Send a prompt to OpenAI and return the text response."""
    client = openai.OpenAI()
    response = client.chat.completions.create(
        model=model,
        messages=[{"role": "user", "content": prompt}],
        max_tokens=max_tokens
    )
    return response.choices[0].message.content

def build_messages(system_prompt: str, history: list, user_input: str) -> list:
    """Build the messages array for a chat completion."""
    messages = [{"role": "system", "content": system_prompt}]
    messages.extend(history)
    messages.append({"role": "user", "content": user_input})
    return messages

# Usage
answer = ask_llm("What is the capital of France?")
print(answer)

# With custom params
summary = ask_llm(
    prompt="Summarize this: [article text]",
    model="gpt-4o-mini",
    max_tokens=200
)

Working with Files

AI workflows constantly read data files and write results. Here are the patterns you'll use most.

import json
import csv

# Read a text file
with open("document.txt", "r", encoding="utf-8") as f:
    text = f.read()

# Read CSV data (the most common data format)
rows = []
with open("customers.csv", "r") as f:
    reader = csv.DictReader(f)  # Each row becomes a dict
    for row in reader:
        rows.append(row)

# rows[0] might look like: {"name": "Alice", "age": "32", "city": "NYC"}

# Read JSON (API configs, embeddings, results)
with open("config.json", "r") as f:

# Write JSON results
results = [{"text": "Hello", "sentiment": "positive", "score": 0.98}]
with open("results.json", "w") as f:
    json.dump(results, f, indent=2)

# Append to a log file
with open("errors.log", "a") as f:
    f.write(f"ERROR: API timeout at {timestamp}\n")

Calling AI APIs

The pattern for calling LLM APIs is the same everywhere — install the SDK, create a client, send messages, parse the response.

import anthropic  # pip install anthropic

client = anthropic.Anthropic(api_key="your-key-here")
# Or set env variable: export ANTHROPIC_API_KEY="your-key"

# Simple call
message = client.messages.create(
    model="claude-3-5-sonnet-20241022",
    max_tokens=1024,
    messages=[
        {"role": "user", "content": "Explain backpropagation simply."}
    ]
)
print(message.content[0].text)

# With system prompt
message = client.messages.create(
    model="claude-3-5-sonnet-20241022",
    max_tokens=1024,
    system="You are a Python tutor. Use code examples in every answer.",
    messages=[
        {"role": "user", "content": "How do list comprehensions work?"}
    ]
)

# Error handling — always wrap API calls
import time

def safe_llm_call(prompt: str, retries: int = 3) -> str:
    for attempt in range(retries):
        try:
            return ask_llm(prompt)
        except Exception as e:
            if attempt < retries - 1:
                time.sleep(2 ** attempt)  # exponential backoff
            else:
                raise e

Quick Check

❓ What does messages[-1] return?▼
The last item in the list. Python negative indexing counts from the end. [-1] = last, [-2] = second to last, etc. This is extremely useful for getting the most recent message in a chat history.
❓ How do you safely access a dict key that might not exist?▼
Use .get(key, default) — e.g., response.get("error", None). If the key doesn't exist, it returns the default instead of raising a KeyError.

Lab: Read a CSV and Summarize Data with AI

Estimated time: 45 minutes · Google Colab (free GPU included)

Open in Google Colab

You'll write Python code to: load a CSV dataset of student grades, compute statistics, then use an LLM to generate a natural-language summary for a teacher.

1

Load and Explore the Dataset

import csv
import json

# Load the student dataset
students = []
with open("students.csv", "r") as f:
    for row in csv.DictReader(f):
        students.append({
            "name": row["name"],
            "grade": int(row["grade"]),
            "subject": row["subject"]
        })

print(f"Loaded {len(students)} student records")
print("First record:", students[0])
2

Compute Statistics

# Calculate summary statistics
grades = [s["grade"] for s in students]
avg_grade = sum(grades) / len(grades)
passing = [g for g in grades if g >= 60]
pass_rate = len(passing) / len(grades) * 100

# Group by subject
from collections import defaultdict
by_subject = defaultdict(list)
for s in students:
    by_subject[s["subject"]].append(s["grade"])

subject_avgs = {subj: sum(gs)/len(gs) for subj, gs in by_subject.items()}

print(f"Average grade: {avg_grade:.1f}")
print(f"Pass rate: {pass_rate:.1f}%")
print(f"Subject averages: {subject_avgs}")
3

Generate AI Summary

import anthropic

client = anthropic.Anthropic(api_key="YOUR_KEY")

stats_text = f"""
Student Dataset Summary:
- Total students: {len(students)}
- Average grade: {avg_grade:.1f}/100
- Pass rate: {pass_rate:.1f}%
- Subject averages: {json.dumps(subject_avgs, indent=2)}
- Lowest grade: {min(grades)}
- Highest grade: {max(grades)}
"""

prompt = f"""
You are a school data analyst.
Write a 3-paragraph summary of these student performance statistics for a teacher:
{stats_text}

Include: overall performance, subject insights, and 2 specific recommendations.
"""

response = client.messages.create(
    model="claude-3-5-sonnet-20241022",
    max_tokens=500,
    messages=[{"role": "user", "content": prompt}]
)

print(response.content[0].text)

Cost Optimization

At scale, token costs compound fast. The difference between a well-optimized and naively implemented AI app can be 10–100x in cost.

import anthropic
import hashlib
import json
from functools import lru_cache

client = anthropic.Anthropic()

# 1. CACHING — identical inputs should never hit the API twice
class ResponseCache:
    def __init__(self):
        self._cache = {}

    def _key(self, model: str, messages: list, system: str) -> str:
        payload = json.dumps({"model": model, "messages": messages, "system": system}, sort_keys=True)
        return hashlib.sha256(payload.encode()).hexdigest()

    def get(self, model, messages, system=""):
        return self._cache.get(self._key(model, messages, system))

    def set(self, model, messages, system, response):
        self._cache[self._key(model, messages, system)] = response

cache = ResponseCache()

def cached_call(model: str, messages: list, system: str = "", max_tokens: int = 512) -> str:
    cached = cache.get(model, messages, system)
    if cached:
        print("[CACHE HIT]")
        return cached

    response = client.messages.create(
        model=model, max_tokens=max_tokens,
        system=system, messages=messages
    )
    result = response.content[0].text
    cache.set(model, messages, system, result)
    return result

# 2. MODEL ROUTING — use cheaper models for simple tasks
def route_to_model(query: str) -> str:
    """Route simple queries to Haiku, complex ones to Sonnet/Opus."""
    simple_indicators = ["what is", "define", "who is", "when did", "how many"]
    is_simple = any(query.lower().startswith(ind) for ind in simple_indicators)

    if is_simple:
        return "claude-haiku-4-5-20251001"   # ~20x cheaper than Opus
    return "claude-sonnet-4-6"               # Good balance

# 3. PROMPT CACHING — for long static system prompts (Anthropic feature)
def call_with_prompt_cache(static_context: str, user_query: str) -> str:
    """Cache the static context prefix to save tokens on repeated calls."""
    response = client.messages.create(
        model="claude-sonnet-4-6",
        max_tokens=512,
        system=[{
            "type": "text",
            "text": static_context,
            "cache_control": {"type": "ephemeral"}  # Cache this prefix
        }],
        messages=[{"role": "user", "content": user_query}]
    )
    return response.content[0].text

Observability and Logging

You can't debug what you can't see. Log every request: inputs, outputs, latency, tokens, cost, and errors. This is non-negotiable in production.

import time
import logging
import uuid
from dataclasses import dataclass, asdict

logging.basicConfig(level=logging.INFO, format='%(asctime)s %(levelname)s %(message)s')
logger = logging.getLogger("ai_app")

@dataclass
class LLMCallLog:
    request_id: str
    model: str
    input_tokens: int
    output_tokens: int
    latency_ms: float
    cost_usd: float
    cache_hit: bool
    error: str | None

COST_PER_1M = {
    "claude-opus-4-7": {"input": 15.0, "output": 75.0},
    "claude-sonnet-4-6": {"input": 3.0, "output": 15.0},
    "claude-haiku-4-5-20251001": {"input": 0.25, "output": 1.25},
}

def observed_call(model: str, messages: list, system: str = "", max_tokens: int = 512) -> tuple[str, LLMCallLog]:
    request_id = str(uuid.uuid4())[:8]
    start = time.time()
    error = None
    result = ""
    input_tokens = output_tokens = 0

    try:
        response = client.messages.create(
            model=model, max_tokens=max_tokens,
            system=system, messages=messages
        )
        result = response.content[0].text
        input_tokens = response.usage.input_tokens
        output_tokens = response.usage.output_tokens
    except Exception as e:
        error = str(e)
        logger.error(f"[{request_id}] LLM call failed: {e}")

    latency_ms = (time.time() - start) * 1000
    costs = COST_PER_1M.get(model, {"input": 0, "output": 0})
    cost = (input_tokens / 1_000_000) * costs["input"] + (output_tokens / 1_000_000) * costs["output"]

    log = LLMCallLog(request_id=request_id, model=model, input_tokens=input_tokens,
                     output_tokens=output_tokens, latency_ms=latency_ms,
                     cost_usd=cost, cache_hit=False, error=error)

    logger.info(f"[{request_id}] model={model} tokens={input_tokens}+{output_tokens} latency={latency_ms:.0f}ms cost=${cost:.6f}")
    return result, log

Retries and Rate Limits

LLM APIs have rate limits and occasionally return errors. Production apps need exponential backoff retry logic with jitter.

import time
import random
from anthropic import RateLimitError, APIStatusError

def call_with_retry(model: str, messages: list, system: str = "",
                    max_tokens: int = 512, max_retries: int = 3) -> str:
    """Call LLM with exponential backoff retry on rate limit errors."""
    for attempt in range(max_retries + 1):
        try:
            response = client.messages.create(
                model=model, max_tokens=max_tokens,
                system=system, messages=messages
            )
            return response.content[0].text

        except RateLimitError as e:
            if attempt == max_retries:
                raise

            # Exponential backoff with jitter: 2^attempt * (0.5 to 1.5)
            wait_time = (2 ** attempt) * random.uniform(0.5, 1.5)
            logger.warning(f"Rate limited (attempt {attempt+1}/{max_retries}). Waiting {wait_time:.1f}s")
            time.sleep(wait_time)

        except APIStatusError as e:
            if e.status_code in {500, 529} and attempt < max_retries:
                wait_time = (2 ** attempt) * random.uniform(0.5, 1.5)
                logger.warning(f"Server error {e.status_code}, retrying in {wait_time:.1f}s")
                time.sleep(wait_time)
            else:
                raise

    raise RuntimeError("Max retries exceeded")

Rate Limiting Users

Protect your API costs from abuse by rate-limiting requests per user.

from collections import defaultdict
import time

class UserRateLimiter:
    """Token bucket rate limiter per user."""
    def __init__(self, requests_per_minute: int = 10):
        self.limit = requests_per_minute
        self.windows = defaultdict(list)  # user_id → list of timestamps

    def is_allowed(self, user_id: str) -> bool:
        now = time.time()
        window = self.windows[user_id]

        # Remove timestamps older than 60 seconds
        self.windows[user_id] = [t for t in window if now - t < 60]

        if len(self.windows[user_id]) >= self.limit:
            return False

        self.windows[user_id].append(now)
        return True

limiter = UserRateLimiter(requests_per_minute=10)

def handle_request(user_id: str, message: str) -> dict:
    if not limiter.is_allowed(user_id):
        return {"error": "Rate limit exceeded. Please wait before sending another message.", "status": 429}

    response = call_with_retry("claude-sonnet-4-6", [{"role": "user", "content": message}])
    return {"response": response, "status": 200}
